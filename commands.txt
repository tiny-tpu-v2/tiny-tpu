# DE1-SoC + Arduino MNIST JTAG quick commands (run from ~/tiny-tpu)

# 0) Go to repo root
cd ~/tiny-tpu

# 1) Full automated setup + continuous inference loop
# Handles: Arduino attach to WSL, serial permission fix, FPGA program, JTAG health, one-shot predict, then live loop.
bash de1_soc_mnist_demo/quickstart_jtag_demo.sh

# 1b) Faster per-frame loop (disable MMIO writeback verification)
bash de1_soc_mnist_demo/quickstart_jtag_demo.sh --no-verify-writeback

# 2) Setup/smoke test only (no continuous loop)
bash de1_soc_mnist_demo/quickstart_jtag_demo.sh --no-loop

# 3) If both boards were unplugged/replugged, run this again
# It retries setup steps and re-establishes the full pipeline.
bash de1_soc_mnist_demo/quickstart_jtag_demo.sh --setup-retries 15 --retry-delay-s 2

# 4) If you rebuilt hardware and need a fresh bitstream first
bash de1_soc_mnist_demo/quickstart_jtag_demo.sh --build

# 5) Optional: run loop without one-shot sanity check
bash de1_soc_mnist_demo/quickstart_jtag_demo.sh --skip-oneshot
